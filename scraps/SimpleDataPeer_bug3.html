Open the JS console, then here's some things to type to play with SimpleDataPeer which wraps PeerConnection's data channels into a simpler Connection data channels...
var peerA, peerB; peerA = new SimpleDataPeer({debug: true, debugPeerName: "p1_alice"}); peerB = new SimpleDataPeer({debug: true, debugPeerName: "p1_bob"}); peerA.setSendSignalMessage(peerB.handleSignalMessage.bind(peerB)); peerB.setSendSignalMessage(peerA.handleSignalMessage.bind(peerA)); peerA.newDataChannel("a"); peerA.newDataChannel("b"); peerA.newDataChannel("c");
