Open the JS console, then here's some things to type to play with SimpleDataPeer which wraps PeerConnection's data channels into a simpler Connection data channels...
var peerA, peerB; peerA = new DataPeer({debug: true, debugPeerName: "p1_alice"}); peerB = new DataPeer({debug: true, debugPeerName: "p1_bob"}); peerA.setSendSignalMessage(peerB.handleSignalMessage.bind(peerB)); peerB.setSendSignalMessage(peerA.handleSignalMessage.bind(peerA)); peerA.newDataChannel("a"); peerB.send("b","hello?"); setTimeout(function() { peerA.newDataChannel("c"); peerB.send("b", "after 1/2 second on b"); }, 500); setTimeout(function() { peerA.send("a", "after 1 second on a"); peerA.send("c", "after 1 second on c"); }, 1000);
